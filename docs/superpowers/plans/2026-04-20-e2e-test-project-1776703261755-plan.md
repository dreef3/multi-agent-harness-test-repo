Tasks

Task 1 — Create analysis script
- Path: scripts/generate-codebase-analysis.sh
- Purpose: Run a deterministic set of commands to inspect the repository structure and produce two outputs: docs/codebase-analysis.md and docs/codebase-analysis-raw.txt.
- Commands to include (exact shell snippet):

  #!/usr/bin/env bash
  set -euo pipefail
  out_md="docs/codebase-analysis.md"
  out_raw="docs/codebase-analysis-raw.txt"
  mkdir -p docs

  echo "Repository: $(git rev-parse --show-toplevel 2>/dev/null || echo .)" > "$out_raw"
  echo "---" >> "$out_raw"

  echo "Top-level list:" >> "$out_raw"
  ls -la --group-directories-first . >> "$out_raw" 2>&1 || true

  echo "\nTree (depth=2):" >> "$out_raw"
  if command -v tree >/dev/null 2>&1; then
    tree -L 2 -a --dirsfirst >> "$out_raw" 2>&1 || true
  else
    find . -maxdepth 2 -mindepth 1 -print | sed 's|^./||' >> "$out_raw" 2>&1 || true
  fi

  echo "\nDetect common manifests:" >> "$out_raw"
  for f in package.json pyproject.toml requirements.txt Pipfile setup.py go.mod Cargo.toml pom.xml build.gradle build.gradle.kts Makefile Dockerfile; do
    if [ -f "$f" ]; then
      echo "- Found $f" >> "$out_raw"
    fi
  done

  echo "\nSearch for CI and workflow files:" >> "$out_raw"
  if [ -d .github ]; then
    echo "- .github directory present" >> "$out_raw"
    ls -la .github || true
  fi
  for f in .travis.yml .gitlab-ci.yml .circleci Dockerfile; do
    if [ -f "$f" ]; then
      echo "- Found $f" >> "$out_raw"
    fi
  done

  echo "\nLanguages (counts):" >> "$out_raw"
  if command -v rg >/dev/null 2>&1; then
    rg --hidden --no-ignore -g '!node_modules' --files | sed -n 's/.*\.\([[:alnum:]]\+\)$/\1/p' | sort | uniq -c | sort -rn >> "$out_raw" 2>&1 || true
  else
    find . -type f | sed -n 's/.*\.\([[:alnum:]]\+\)$/\1/p' | sort | uniq -c | sort -rn >> "$out_raw" 2>&1 || true
  fi

  echo "\nQuick grep for test directories and files:" >> "$out_raw"
  for t in tests spec test __tests__; do
    find . -type d -name "$t" -print -prune >> "$out_raw" 2>&1 || true
  done

  # Generate the human readable markdown using the raw file as source
  {
    echo "# Codebase Analysis Report"
    echo ""
    echo "Generated: $(date -u +'%Y-%m-%dT%H:%M:%SZ')"
    echo ""
    echo "## Summary"
    echo "- Repository: $(basename "$(pwd)")"
    echo "- Analysis script: scripts/generate-codebase-analysis.sh"
    echo ""
    echo "## Highlights (automatically detected)"
    echo '```'
    sed -n '1,200p' "$out_raw"
    echo '```'
    echo ""
    echo "## Recommendations"
    echo "- Add or update README with clear build/test instructions if missing."
    echo "- Add a CI badge in README if CI workflows detected."
    echo "- Consider adding CODEOWNERS or CONTRIBUTING.md if missing."
  } > "$out_md"

  echo "Analysis written to $out_md and $out_raw"

- Notes: The script is intentionally read-only and avoids running builds or tests.
